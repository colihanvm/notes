1function [centers] = detectCircles(im, edges, radius, top_k, ~)
 2
 3centers = zeros(top_k, 2);
 4quant_val = 5;
 5H = zeros(size(im,1), size(im,2));
 6
 7for i=1:size(edges,1) %over every edge
 8    %[x, y, mag, ori] = edges(i, :);
 9    x = edges(i, 1); y = edges(i, 2); theta = edges(i, 4);
10
11    a = x - radius * cosd(theta);
12    b = y - radius * sind(theta);
13    a = ceil(a/quant_val);
14    b = ceil(b/quant_val);
15    if (a>0 && b>0)
16        if a < size(H,1) && b < size(H,2)
17            H(a,b) = H(a,b) + 1;
18        end
19    end 
20end
21
22H_sort = sort(H(:), 'descend');
23repeat = 2;
24for i=1:top_k
25    [row, col] = find(H == H_sort(i));
26    if (i > 1 && H_sort(i) == H_sort(i-1))
27        centers(i,1) = row(repeat);
28        centers(i,2) = col(repeat);
29        repeat = repeat + 1;
30    else
31        repeat = 2;
32        centers(i,1) = row(1);
33        centers(i,2) = col(1);
34    end 
35end
36centers = centers*quant_val;
