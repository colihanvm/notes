1function [im_chunk] = process_stanza(im_chunk)
 2
 3%original chunk
 4% figure; imshow(im_chunk)
 5
 6%calculate column sum over the stanza
 7tmp = sum(double(im_chunk), 1)';
 8%  figure; plot(tmp);  title('col sum');
 9
10%find value to threshold with
11mean_val = sum(tmp)/size(tmp,1);
12std_val = std(tmp);
13threshold_val = mean_val - 2 * std_val;
14
15%select vertical bits
16ind = tmp < threshold_val;
17
18%delete vertical bits
19im_chunk(:, ind) = 255;
20
21%find nasty flags
22max_val = tmp(1); %max(tmp);
23del = 0;
24last = -1;
25same = 0;
26for i=1:size(ind,1)-1
27    if (del) %last index was deleted or was rightmost edge
28        if (tmp(i) < max_val)
29            if (same)
30                if (tmp(i) ~= last)
31                    del = 0;
32                end
33            end
34            if (tmp(i) == last)
35                same = 1;
36            end
37            im_chunk(:, i) = 255;
38            last = tmp(i);
39        else
40            del = 0;
41        end
42        
43    elseif (ind(i) && ~ind(i+1)) %find rightmost edge of a vert line
44        del = 1;
45        same = 0;
46    end
47end
48
49%the result!
50% figure; imshow(im_chunk)
51end
