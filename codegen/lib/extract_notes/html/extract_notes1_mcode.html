  3        %     radius = 7; %pretty good for 4th, 8th
  4    %     num = 60; 
  5    % threshold = 155;
  6    % im = imread('rowrowbt_mod3.png');
  7    im = rgb2gray(im);
  8
  9    for i=1:size(im,1)
 10        for j=1:size(im,2)
 11            if im(i,j) > 240
 12                im(i,j) = 255;
 13            end
 14        end
 15    end
 16    % figure; imshow(im); title('pixel color adjust');
 17

 19    tmp = sum(double_im, 2);
 20    % figure; plot(tmp);  title('row sum');
 21    % 
 22    % pause;
 23
 24    ind = find(tmp < 147000);
 25    im(ind, :) = 255;
 26    % figure; imshow(im); title('lines erased');
 27
 28    %pause;
 29
 30
 31    %calculate staff lines
 32    staffs = zeros(10,1);
 33    staffs(1) = ind(1);
 34    lines = zeros(size(ind));
 35    j = 1;
 36    num = 1;
 37    lines(1) = 1;
 38    staff_num = zeros(size(staffs));
 39    staff_num(1) = 1;
 40    for i=2:size(ind,1)
 41        if (ind(i) > (ind(i-1) + 1)) %end line deletion
 42            staffs(j) = floor(staffs(j) / num);
 43            num = 1;

 45            staffs(j) = ind(i); %first line
 46            lines(i) = j;
 47            staff_num(j) = staff_num(j) + 1;
 48        else
 49            staffs(j) = staffs(j) + ind(i); %there was a line deleted above this
 50            lines(i) = j;
 51            staff_num(j) = staff_num(j) + 1;
 52            num = num + 1;
 53            if i == size(ind,1) %last
 54                staffs(j) = floor(staffs(j) / num);
 55            end 
 56        end
 57    end
 58
 59    % figure; imshow(im); title('visualization of no staff lines'); hold on

 67    for s=1:size(staffs,1) %y values of staffs. s = index of staffs AKA staff number
 68        above_s = ind(lines_passed)-1;
 69        below_s = ind(lines_passed) + staff_num(s); %ind(lines_passed) + staff_num(s)-1 + 1;
 70        for x=1:size(im,2) %x pos of image
 71
 72            if (im(above_s, x) < 200)
 73                if(im(below_s, x) < 200)
 74                    im(ind(lines_passed):(ind(lines_passed + staff_num(s) - 1)), x) = floor((im(above_s, x) + im(below_s, x))/2);
 75                elseif (im(above_s, x) < 150)
 76                    im(ind(lines_passed):(ind(lines_passed + staff_num(s) - 1)), x) = floor((im(above_s, x))/2);
 77                end
 78            elseif(im(below_s, x) < 150)
 79                im(ind(lines_passed):(ind(lines_passed + staff_num(s) - 1)), x) = floor((im(below_s, x))/2);
 80            end
 81
 82        end
 83        lines_passed = lines_passed + staff_num(s);
 84    end
 85
 86
 87    % figure; imshow(im); title('visualization of slightly more lines'); hold on
 88
 89
 90
 91    % figure; imshow(im); title('visualization of staff lines'); hold on

 95

103    [~, in] = sort(centers(:,2));
104    centers = centers (in,:);

106    %also check for stacked circles (works if no chords)
107
108
109    % NEXT PART:
110
111    %sort for x pos
112    [~, in] = sort(centers(:,1));
113    centers = centers (in,:);
114
115    %cleanup
116    % centers(6,:) = [];
117    % centers(7:8,:) = [];
118    % centers(8,:) = [];
119    % centers(12,:) = [];
120
121
122    resort = zeros(size(centers,1), 1);
123    temp = zeros(size(centers));
124
125    for i=1:size(centers,1)
126        if centers(i,2) > (staffs(5) + staffs(6))/2
127            resort(i) = 1;
128        end
129    end
130    %reorder staff line notes so they floooow
131    temp(resort == 1, :) = centers(resort == 1, :);
132    temp(resort == 0, :) = [];

135    centers = [centers; temp];

137
138
139    scale = ['F';'D';'B';'G';'E'];
140    scale = repmat(scale, size(staffs,1)/size(scale,1), 1);
141

144    % plot(x, staffs*ones(size(x))); hold on
145
146

149    %    viscircles(centers(i,:), radius * ones(size(centers(i), 1), 1)); hold on
150         int_notes(i,:)=closest_to(centers(i,2),staffs,scale);

156    % plot(x, staffs*ones(size(x)));
157end
